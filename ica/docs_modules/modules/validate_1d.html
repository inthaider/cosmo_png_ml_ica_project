118def calculate_residuals_ica(x, y, norm=True, relative=True):
119    r"""Compute both scalar & vector residuals between $x$ & $y$, where $x$ is the source field and $y$ is the estimated field.
120
121    Scalar residual $r$ is defined as:
122        $r_s = 1 - \frac{\left\|a\right\|_2}{\left\|b\right\|_2}$
123    Vector residual $r_v$ is defined as:
124        $r_v = 1 - \frac{\left\|a\right\|_2}{\left\|b\right\|_2}$
125    where $\left\|a\right\|_2$ is the 2-norm of $x$ and $\left\|b\right\|_2$ is the 2-norm of $y$.
126
127    Parameters
128    ----------
129    x : np.ndarray
130        True field $x$. Must be same shape as $y$.
131    y : np.ndarray
132        Extracted field $y$. Must be same shape as $x$.
133        
134    Returns
135    -------
136    rs : float
137        Scalar residual of $x$ and $y$ (see below).
138    rv : np.ndarray
139        Vector residual of $x$ and $y$ (see below).
140
141    Notes
142    -----
143        (1) Normalize vector y by mean-subtraction and standard deviation-division, then rescale it by the standard deviation of x. 
144        This step ensures that y has the same mean and scale as x, which is essential for a fair comparison.
145    
146        (2) The vector residual (rv) is calculated by projecting the rescaled y vector onto the x vector, then dividing the result by the magnitude of the x vector. 
147        This measures how well y aligns with x, considering both direction and magnitude.
148        The vector residual ($rv$) is calculated as:
149            rv = 1 - ( ( b.a / a.a ) * a ) / |a|
150
151        (3) The scalar residual (rs) is calculated by finding the magnitude of the projection of y onto x, dividing it by the square of the magnitude of x, and subtracting the result from 1. 
152        This measures the absolute difference between the magnitudes of: x AND the projection of y onto x.
153        The scalar residual ($rr$) is calculated as:
154            rs = 1 - | ( b.a / |a| ) / |a| |
155
156
157        Note that the magnitude of x is:
158            |x| = (x.x)^{1/2}
159        where $x$ is a vector.
160
161    TODO
162    ----
163    FILL IN WHY IT MAKES SENSE TO CHOOSE TO COMPUTE $RV$ & $RR$ THIS WAY!
164    """
165    if x is None or y is None:
166        raise ValueError("Invalid input field(s): Field must be a vector/array, not None.")
167    if len(x) == 0 or len(y) == 0:
168        raise ValueError("Invalid input field(s): Field must have a nonzero size/length.")
169
170    if norm:
171        # Normalize y by std-division and rescale by x's std
172        y = rescale_extracted_field(x, y)
173        
174        # print("y_normd: ", y)
175
176    # Calculate dot products and scalar projection
177    ydotx = np.dot(y, x)
178    xdotx = np.dot(x, x)
179    mag_x = np.sqrt(xdotx)      # magnitude of x: |x| = (x.x)^{1/2}
180    scalar_proj_xy = ydotx/mag_x
181    
182    # print("ydotx: ", ydotx)
183    # print("xdotx: ", xdotx)
184    # print("proj: ", scalar_proj_xy)
185    # print("proj/|x|: ", scalar_proj_xy/mag_x)
186
187    # Calculate scalar residuals
188    if not relative:
189        # Calculate absolute scalar residual
190        rs = 1 - scalar_proj_xy
191        
192        # print("rs (1 - proj): ", rs)
193    else:
194        # Calculate relative scalar residual
195        rs_test = 1 - (scalar_proj_xy / mag_x)        # rs = 1 - (b.a / |a|) / |a|
196        rs = 1 - np.abs(scalar_proj_xy) / mag_x        # rs = 1 - |(b.a / |a|) / |a||; absolute value to ensure positive scalar projection
197
198        # print("rs, not absolute ( 1 - proj/|x| ): ", rs_test)
199        # print("rs ( 1 - |(proj/|x|)| ): ", rs)
200
201    print("rs_ica: ", rs)
202
203    return np.abs(rs)       # return absolute value of scalar residual since we want the relative scalar residual to be positive

334def calculate_residuals(x, y, norm=True, relative=True):
335    r"""Compute both scalar & vector residuals between $x$ & $y$, where $x$ is the source field and $y$ is the estimated field.
336
337    The residual (b - <b\cdot a><a\cdot a>^-1 a) or its magnitude, an associated quadratic. Here a is the input nonG and b is the output nonG, and if it was perfect separation this would be zero.
338
339    Parameters
340    ----------
341    x : np.ndarray
342        True field $x$. Must be same shape as $y$.
343    y : np.ndarray
344        Extracted field $y$. Must be same shape as $x$.
345        
346    Returns
347    -------
348    rs : float
349        Scalar residual of $x$ and $y$ (see below).
350    rv : np.ndarray
351        Vector residual of $x$ and $y$ (see below).
352
353    Notes
354    -----
355    """
356    if x is None or y is None:
357        raise ValueError("Invalid input field(s): Field must be a vector/array, not None.")
358    if len(x) == 0 or len(y) == 0:
359        raise ValueError("Invalid input field(s): Field must have a nonzero size/length.")
360
361    if norm:
362        # Normalize y by mean-subtraction and std-division and rescale by x's std
363        y = rescale_extracted_field(x, y)
364        
365        # print("y_normd: ", y)
366
367    # Calculate dot products and vector projection
368    ydotx = np.dot(y, x)
369    xdotx = np.dot(x, x)
370    mag_x = np.sqrt(xdotx)      # magnitude of x: |x| = (x.x)^{1/2}
371    vector_proj_xy = (ydotx / mag_x**2) * x     # (( y.x / |x|^2 ) * x) gives the projection of y onto x; |x|^2 is equivalent to x.x
372
373    # print("ydotx: ", ydotx)
374    # print("xdotx: ", xdotx)
375
376    # Calculate residuals
377    if not relative:        
378        # Calculate absolute residuals
379        rv = y - vector_proj_xy      # absolute vector residual; subtracting from y gives the vector pointing from the proj_{x}y (proj of y onto x) to y
380        rs = np.sqrt(np.dot(rv, rv))        # absolute scalar residual; magnitude of vector residual
381    else:       
382        # Calculate relative residuals, where the vector projection is calculated in units of |x|
383        rv = ( y - vector_proj_xy ) / mag_x     # relative vector residual, since we're dividing by |x|
384        # rv = 1 - ( (ydotx / xdotx) * x ) / xmag         # different form of relative vector residual; here, rv = 1 - (( y.x / x.x ) * x ) / |x|; subtracting from 1 gives an inverted residual (i.e. higher values indicate a better match)
385        rs = np.sqrt(np.dot(rv, rv))        # relative scalar residual (since rv is relative)
386        
387        # test_rv = y - vector_proj_xy
388        # test_rs = np.sqrt(np.dot(test_rv, test_rv)) / mag_x
389        # print("test_rs: ", test_rs)
390
391    print("rs: ", rs)
392    # print("rv: ", rv)
393
394    return np.abs(rs), rv       # return absolute value of scalar residual since we want the scalar residual to be positive

503def calculate_pearson_coefficient(x, y):
504    r"""Calculate Pearson correlation coefficient between two fields, where $x$ is the source field and $y$ is the estimated field.
505
506    The Pearson correlation coefficient is a measure of the linear correlation between two variables $X$ and $Y$.
507    It has a value between -1 and 1, where 1 is total positive linear correlation, 0 is no linear correlation, and -1 is total negative linear correlation.
508    The mathematical definition of the coefficient is:
509        $r = \frac{\sum_{i=1}^{n} (x_i - \bar{x})(y_i - \bar{y})}{\sqrt{\sum_{i=1}^{n} (x_i - \bar{x})^2 \sum_{i=1}^{n} (y_i - \bar{y})^2}}$
510    where $\bar{x}$ is the mean of the data in $X$ and $\bar{y}$ is the mean of the data in $Y$.
511    The coefficient is also known as the Pearson product-moment correlation coefficient.
512    
513    Parameters
514    ----------
515    x : np.ndarray
516        The true field.
517    y : np.ndarray
518        The extracted field.
519
520    Returns
521    -------
522    correlation_coefficient : float
523
524    Examples
525    --------

547    r"""Calculate the biweight midcorrelation between two fields, where $x$ is the source field and $y$ is the estimated field.
548    
549    Biweight midcorrelation is a robust correlation measur between two variables $X$ and $Y$ based on the biweight midvariance, which is a robust alternative to the classical variance. It calculates the correlation between two variables while down-weighting the contribution of outliers. The biweight midcorrelation ranges from -1 to 1, similar to the Pearson correlation coefficient, with 1 indicating a perfect positive relationship, -1 a perfect negative relationship, and 0 no relationship.
550    The mathematical definition of the biweight midcorrelation is:
551        $r_{bc} = \frac{\sum_{i=1}^{n} (x_i - \tilde{x})(y_i - \tilde{y})}{\sqrt{\sum_{i=1}^{n} (x_i - \tilde{x})^2 \sum_{i=1}^{n} (y_i - \tilde{y})^2}}$
552

557    y : np.ndarray
558        The extracted field.

564

568        >>> y = np.array([1.2, 2.1, 2.9, 4.1, 5.2, 101])  # Extracted/estimated field
569        
570        >>> result = biweight_midcorrelation(x, y)
571        >>> print(f"Biweight midcorrelation: {result:.2f}")
572
573    Notes
574    -----
575        Keep in mind that this method assumes that your data is reasonably well-behaved (i.e., the majority of the data points are not outliers), as extreme cases with a large number of outliers can still affect the biweight midcorrelation. If you think your data might have a high percentage of outliers or other peculiarities, consider exploring other robust correlation methods as well.
576    """
577    if x is None or y is None:
578        raise ValueError("Invalid input field(s): Field must be a vector/array, not None.")
579    if len(x) == 0 or len(y) == 0:
580        raise ValueError("Invalid input field(s): Field must have a nonzero size/length.")

583    y = rescale_extracted_field(x, y)
584    
585    # Calculate the medians of the input arrays
586    # These are used as central values to compute the biweight midvariances
587    median_x = np.median(x)
588    median_y = np.median(y)
589
590    # Calculate the scaling factors for each input array (x and y)
591    # The scaling factor is based on the median absolute deviation (MAD)
592    # The MAD is multiplied by 9 to obtain a more robust estimate of the spread
593    scaling_factor_x = 9 * np.median(np.abs(x - median_x))
594    scaling_factor_y = 9 * np.median(np.abs(y - median_y))
595
596    # Compute the u values for each input array
597    # u is the scaled distance of each data point from the median
598    u_x = (x - median_x) / scaling_factor_x
599    u_y = (y - median_y) / scaling_factor_y

601    # Calculate the biweight midcorrelation
602    # We only consider data points with |u| < 1 to reduce the impact of outliers
603    mask = (np.abs(u_x) < 1) & (np.abs(u_y) < 1)

605    # Calculate the numerator of the biweight midcorrelation formula
606    # This is the sum of the product of the differences between each data point and the median
607    # The product is weighted by (1 - u^2)^2 for both x and y to down-weight the contribution of outliers
608    numerator = np.sum((x[mask] - median_x) * (y[mask] - median_y) * (1 - u_x[mask]**2)**2 * (1 - u_y[mask]**2)**2)
609    
610    # Calculate the denominator of the biweight midcorrelation formula
611    # The denominator is the product of the sum of squared differences between each data point and the median
612    # The sum is weighted by (1 - u^2)^2 for both x and y to down-weight the contribution of outliers
613    denominator_x = np.sum((x[mask] - median_x)**2 * (1 - u_x[mask]**2)**2)
614    denominator_y = np.sum((y[mask] - median_y)**2 * (1 - u_y[mask]**2)**2)
615
616    # Calculate the biweight midcorrelation by dividing the numerator by the square root of the product of the denominators
617    correlation = numerator / np.sqrt(denominator_x * denominator_y)
